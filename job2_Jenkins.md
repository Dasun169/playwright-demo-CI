# Jenkins Job 2: Run Tests & Generate Reports

This document explains the mechanism of **Job 2** in our Jenkins pipeline, responsible for running the Playwright tests inside the deployed Docker container and generating execution reports.

## Pipeline Mechanism

Job 2 focuses on executing tests based on runtime parameters chosen by the user:

1. **Setup Workspace**: Removes and recreates required execution directories (`logs`, `allure-results`, `test-reports`) to ensure testing occurs in a completely clean workspace.
2. **Pull Docker Image**: Pulls the Docker image built by Job 1 from Docker Hub into the Jenkins execution environment.
3. **Run Tests**: Executes playwright tests dynamically passing the `PROJECT` parameter (e.g., Firefox, WebKit) and `TEST_TAG` (`@smoke`, etc.). The directories generated by step 1 are mounted as Docker volumes so results flow directly back to the Jenkins host.
4. **Generate Allure Report**: Collects the `allure-results` data outputted by the container and invokes the Allure Jenkins plugin to build an engaging HTML interactive report.
5. **Archive Test Results**: Condenses all logs, reports, and Allure data and archives them within Jenkins as fingerprinted artifacts for persistence.
6. **Post Actions**: Enforces cleanup of the Docker images via removal and pruning to ensure clean slave nodes for future jobs, and correctly relays run success or failure info.

## Jenkinsfile (Groovy)

```groovy
pipeline {
    agent any
    
    options {
        timeout(time: 30, unit: 'MINUTES')
        buildDiscarder(logRotator(numToKeepStr: '20'))
    }
    
    parameters {
        choice(
            name: 'TEST_TAG',
            choices: ['@smoke', '@regression', '@sanity', ''],
            description: 'Select the test tag to run (leave empty for all tests)'
        )

        choice(
            name: 'PROJECT',
            choices: [
                'chromium',
                'firefox',
                'webkit',
                'Mobile Chrome',
                'Mobile Safari',
                'Microsoft Edge',
                'Google Chrome'
            ],
            description: 'Select the Playwright project to run'
        )
    }
    
    environment {
        DOCKERHUB_REPO = "dasun544/playwright-demo-ci"
        DOCKERHUB_IMAGE_TAG = 'v1.0'
        
        // Use relative paths for Windows
        LOGS_DIR = "logs"
        ALLURE_RESULTS_DIR = "allure-results"
        TEST_REPORTS_DIR = "test-reports"
        
        // Log files
        RAW_LOG = "logs/raw-console.log"
        TEST_LOG = "logs/test-log.log"
        SUMMARY_LOG = "logs/execution-summary.log"
    }
    
    stages {
        stage('Setup') {
            steps {
                script {
                    bat """
                        if exist "${LOGS_DIR}" rmdir /s /q "${LOGS_DIR}"
                        if exist "${ALLURE_RESULTS_DIR}" rmdir /s /q "${ALLURE_RESULTS_DIR}"
                        if exist "${TEST_REPORTS_DIR}" rmdir /s /q "${TEST_REPORTS_DIR}"
                        
                        mkdir "${LOGS_DIR}"
                        mkdir "${ALLURE_RESULTS_DIR}"
                        mkdir "${TEST_REPORTS_DIR}"
                        
                        echo Setup complete
                    """
                }
            }
        }
        
        stage('Pull Docker Image') {
            steps {
                script {
                    def imageTag = "${env.DOCKERHUB_REPO}:${env.DOCKERHUB_IMAGE_TAG}"
                    echo "Pulling Docker image: ${imageTag}"
                    
                    bat """
                        echo Pulling Docker image...
                        docker pull ${imageTag}
                        echo Docker image pulled successfully
                    """
                }
            }
        }
        
        stage('Run Tests') {
            steps {
                script {
                    def imageTag = "${env.DOCKERHUB_REPO}:${env.DOCKERHUB_IMAGE_TAG}"
                    
                    def workspace = pwd()
                    def logsPath = "${workspace}\\${env.LOGS_DIR}".replace('\\', '/')
                    def allurePath = "${workspace}\\${env.ALLURE_RESULTS_DIR}".replace('\\', '/')
                    def reportsPath = "${workspace}\\${env.TEST_REPORTS_DIR}".replace('\\', '/')
                    
                    def testCommand = "npx playwright test --project='${params.PROJECT}'"
                    if (params.TEST_TAG && params.TEST_TAG != '') {
                        testCommand += " --grep='${params.TEST_TAG}'"
                    }
                    testCommand += " --reporter=line,allure-playwright"
                    
                    echo "🎯 Running command: ${testCommand}"
                    
                    bat """
                        echo Running tests in Docker container...
                        
                        docker run --rm ^
                          -v "${logsPath}:/app/logs" ^
                          -v "${allurePath}:/app/allure-results" ^
                          -v "${reportsPath}:/app/test-results" ^
                          -e ALLURE_RESULTS_DIR=/app/allure-results ^
                          ${imageTag} ^
                          sh -c "cd /app && npx playwright install --with-deps && ${testCommand}" > "${RAW_LOG}" 2>&1
                        
                        echo Tests completed with exit code: %errorlevel%
                        echo Raw console log saved to ${RAW_LOG}
                    """
                    
                    bat """
                        if exist "${LOGS_DIR}\\test.log" (
                            copy "${LOGS_DIR}\\test.log" "${TEST_LOG}"
                            echo Test log copied to ${TEST_LOG}
                        ) else (
                            echo No test.log found in logs directory
                        )
                    """
                }
            }
        }
        
        stage('Generate Allure Report') {
            steps {
                script {
                    echo "📊 Generating Allure report..."
                    
                    // Wait a moment for files to be written
                    sleep(2)
                    
                    // SIMPLE APPROACH: Just check if directory exists and has files
                    // Then always try to generate the report - Jenkins Allure plugin will handle empty results
                    
                    // List the files for debugging
                    bat """
                        echo Allure results directory contents:
                        if exist "${ALLURE_RESULTS_DIR}" (
                            dir "${ALLURE_RESULTS_DIR}"
                            echo.
                            echo JSON files found:
                            dir "${ALLURE_RESULTS_DIR}\\*.json" 2>nul || echo No JSON files
                        ) else (
                            echo Allure results directory not found
                        )
                    """
                    
                    // Generate Allure report - always try to generate
                    // The Allure plugin will handle if there are no results
                    allure(
                        includeProperties: false,
                        results: [[path: "${ALLURE_RESULTS_DIR}"]]
                    )
                    
                    echo "✅ Allure report generation attempted"
                }
            }
        }
        
        stage('Archive Test Results') {
            steps {
                script {
                    echo "📦 Archiving test results..."
                    
                    // Create a summary file
                    bat """
                        echo ===== TEST EXECUTION SUMMARY ===== > "${SUMMARY_LOG}"
                        echo Test Tag: ${params.TEST_TAG} >> "${SUMMARY_LOG}"
                        echo Project: ${params.PROJECT} >> "${SUMMARY_LOG}"
                        echo Timestamp: %date% %time% >> "${SUMMARY_LOG}"
                        echo. >> "${SUMMARY_LOG}"
                        
                        if exist "${RAW_LOG}" (
                            echo Raw console log: >> "${SUMMARY_LOG}"
                            echo ------------------------ >> "${SUMMARY_LOG}"
                            type "${RAW_LOG}" >> "${SUMMARY_LOG}"
                        ) else (
                            echo Raw log file not found >> "${SUMMARY_LOG}"
                        )
                        
                        if exist "${TEST_LOG}" (
                            echo. >> "${SUMMARY_LOG}"
                            echo Test log file: >> "${SUMMARY_LOG}"
                            echo ------------------------ >> "${SUMMARY_LOG}"
                            type "${TEST_LOG}" >> "${SUMMARY_LOG}"
                        ) else (
                            echo Test log file not found >> "${SUMMARY_LOG}"
                        )
                        
                        echo. >> "${SUMMARY_LOG}"
                        echo Test Results Directory: >> "${SUMMARY_LOG}"
                        if exist "${TEST_REPORTS_DIR}" (
                            dir "${TEST_REPORTS_DIR}" /s >> "${SUMMARY_LOG}"
                        ) else (
                            echo Test reports directory not found >> "${SUMMARY_LOG}"
                        )
                        
                        echo. >> "${SUMMARY_LOG}"
                        echo Allure Results Directory: >> "${SUMMARY_LOG}"
                        if exist "${ALLURE_RESULTS_DIR}" (
                            dir "${ALLURE_RESULTS_DIR}" >> "${SUMMARY_LOG}"
                        ) else (
                            echo Allure results directory not found >> "${SUMMARY_LOG}"
                        )
                        
                        echo Summary created at ${SUMMARY_LOG}
                    """
                    
                    // Archive all artifacts
                    archiveArtifacts artifacts: """
                        ${LOGS_DIR}/*.log,
                        ${ALLURE_RESULTS_DIR}/*.json,
                        ${ALLURE_RESULTS_DIR}/*.txt,
                        ${ALLURE_RESULTS_DIR}/*.zip,
                        ${TEST_REPORTS_DIR}/**/*
                    """, allowEmptyArchive: true, fingerprint: true
                    
                    // Display summary in console
                    bat """
                        echo.
                        echo ========================================
                        echo TEST EXECUTION SUMMARY
                        echo ========================================
                        if exist "${SUMMARY_LOG}" (
                            type "${SUMMARY_LOG}"
                        ) else (
                            echo Summary file not found
                        )
                        echo ========================================
                        
                        echo.
                        echo Logs directory contents:
                        if exist "${LOGS_DIR}" (
                            dir "${LOGS_DIR}"
                        ) else (
                            echo Logs directory not found
                        )
                        
                        echo.
                        echo Allure results directory contents:
                        if exist "${ALLURE_RESULTS_DIR}" (
                            dir "${ALLURE_RESULTS_DIR}"
                            echo.
                            echo JSON files found:
                            dir "${ALLURE_RESULTS_DIR}\\*.json" 2>nul || echo No JSON files
                        ) else (
                            echo Allure results directory not found
                        )
                        
                        echo.
                        echo Test reports directory contents:
                        if exist "${TEST_REPORTS_DIR}" (
                            dir "${TEST_REPORTS_DIR}"
                        ) else (
                            echo Test reports directory not found
                        )
                    """
                }
            }
        }
    }
    
    post {
        always {
            script {
                echo "🧹 Cleaning up..."
                
                def imageTag = "${env.DOCKERHUB_REPO}:${env.DOCKERHUB_IMAGE_TAG}"
                bat """
                    docker rmi -f ${imageTag} 2>nul || echo Image removal failed or image not found
                    docker system prune -f 2>nul || echo Prune failed
                """
                
                echo "Cleanup completed"
            }
            
            archiveArtifacts artifacts: """
                ${LOGS_DIR}/*.log,
                ${ALLURE_RESULTS_DIR}/*,
                ${TEST_REPORTS_DIR}/**/*
            """, allowEmptyArchive: true
        }
        
        success {
            echo "=============================================="
            echo "✅ Pipeline completed successfully!"
            echo "✅ Tests executed with tag: ${params.TEST_TAG}"
            echo "✅ Project: ${params.PROJECT}"
            echo "✅ Allure report generated"
            echo "=============================================="
        }
        
        failure {
            echo "=============================================="
            echo "❌ Pipeline failed. Check the console output above for details."
            echo "=============================================="
        }
    }
}
```
